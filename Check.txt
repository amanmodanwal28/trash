#!/bin/bash

# Source and destination paths
SRC="$HOME/Downloads/IOB_Server"
DEST="/home/ptcs/iob/bin"

# Check if source exists
if [ -d "$SRC" ]; then
    echo "Moving contents from $SRC to $DEST ..."

    # Create destination if it doesn't exist
    mkdir -p "$DEST"

    # Move all contents (files + folders) from source to destination
    mv "$SRC"/* "$DEST"/

    echo "✅ All contents moved successfully."
else
    echo "❌ Source folder does not exist: $SRC"
fi
